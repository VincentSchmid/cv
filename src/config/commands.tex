% Define the altSection command
\newcommand{\altSection}[1]{%
  {\large\sectionfont\color{altColor} #1\par}
}

% Customparagraph, to have non-bold, gray text
\newcommand{\customparagraph}[1]{{\normalfont #1}}

% Define the dotted line command
\newcommand{\dottedline}{
  \vspace{1em}
  \begin{tikzpicture}
      \draw[dotted, thick, lightgray] (0,0) -- (\textwidth,0);
  \end{tikzpicture}
  \par
  \vspace{1em}
}

% Define a new command for skill logos
\newcommand{\skilllogo}[2]{
  \begin{minipage}[c]{0.9\linewidth}
    \centering
    \includegraphics[width=0.9\linewidth]{assets/logos/#1}
    \vspace{1em}
    \\ \small\textcolor{lightgray}{#2}
  \end{minipage}
}

\newcommand{\mysubsection}[1]{
    {\subsectionfont\fontsize{12}{10}\selectfont\bfseries\color{black} #1\par}
    \vspace{-0.8em}
}

% Define a new command for sections with details
\newcommand{\detailsection}[4]{
    \altSection{#1}
    \par
    \vspace*{0.5em}
    \mysubsection{\href{#3}{#2}}
    \par
    \vspace*{1.5em}
    {\color{black} #4}
    \vspace{0.3em}
}
